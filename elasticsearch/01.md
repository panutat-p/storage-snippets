# Fruits

## Index

```json
PUT /fruits
{
  "mappings": {
    "properties": {
      "name": {
        "type": "text"
      },
      "color": {
        "type": "keyword"
      },
      "price": {
        "type": "integer"
      },
      "tag": {
        "type": "keyword"
      }
    }
  }
}
```

## Search

```json
GET /fruits/_search
```

```json
GET /fruits/_search?filter_path=hits.hits._source
```

```json
GET /fruits/_search
{
  "query": {
    "bool": {
      "filter": [
        {
          "term": {
            "color": "green"
          }
        }
      ]
    }
  }
}
```

## Aggregation

```json
GET /fruits/_search
{
  "size": 0,
  "aggs": {
    "colors": {
      "terms": {
        "field": "color"
      }
    }
  }
}
```

```json
GET /fruits/_search
{
  "size": 0,
  "aggs": {
    "price_filter": {
      "filter": {
        "range": {
          "price": {
            "gt": 10
          }
        }
      },
      "aggs": {
        "colors": {
          "terms": {
            "field": "color"
          }
        }
      }
    }
  }
}
```
