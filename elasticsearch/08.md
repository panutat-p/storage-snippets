# Re-index

```shell
PUT /fruits_old
{
  "settings": {
    "index": {
      "number_of_shards": 1,  
      "number_of_replicas": 1
    }
  },
  "mappings":{
    "dynamic":false,
      "properties": {
        "name": {"type": "text"},
        "color": {"type": "keyword"},
        "price": {"type": "integer"},
        "tags": {"type": "keyword"}
      }
  }
}
```

```shell
PUT /fruits_new
{
  "settings": {
    "index": {
      "number_of_shards": 1,  
      "number_of_replicas": 1
    }
  },
  "mappings":{
    "dynamic":false,
      "properties": {
        "name": {"type": "text"},
        "color": {"type": "text"},
        "price": {"type": "integer"},
        "tags": {"type": "text"}
      }
  }
}
```

```shell
POST _reindex
{
  "source": {
    "index": "fruits_old"
  },
  "dest": {
    "index": "fruits_new"
  }
}
```
